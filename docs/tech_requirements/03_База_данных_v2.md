# Техническое задание: проектирование реляционной базы данных

## Проект: «Ассоциация трихологов»

| Параметр | Значение |
|----------|----------|
| Версия документа | 2.0 |
| Дата | 2026-03-02 |
| Статус | Актуально |
| СУБД | PostgreSQL 16+ |
| Нормализация | 3NF |
| Первичные ключи | UUID v7 (сортируемые по времени) — Python: библиотека `uuid-utils`; SQL: расширение `pg_uuidv7` или функция `gen_uuid_v7()` |
| Временные метки | TIMESTAMPTZ (UTC) |
| Денежные поля | NUMERIC(12,2) |
| Хеширование паролей | Argon2id |
| Генерация slug | `python-slugify` (библиотека) — транслитерация + lowercase + дефисы. Пример: «Иванов Иван» → `ivanov-ivan`. Уникальность обеспечивается добавлением суффикса `-2`, `-3` и т.д. |

## 1. Enum-типы (PostgreSQL CREATE TYPE)

```sql
CREATE TYPE doctor_status AS ENUM (
  'pending_review',  -- заявка отправлена, ожидает проверки
  'approved',        -- модерация пройдена, ожидается оплата
  'rejected',        -- модерация отклонена
  'active',          -- одобрен + оплачен, виден на сайте
  'deactivated'      -- подписка истекла или ручная деактивация
);

CREATE TYPE subscription_status AS ENUM (
  'active',           -- подписка действует
  'expired',          -- подписка истекла
  'pending_payment',  -- ожидает оплаты
  'cancelled'         -- отменена
);

CREATE TYPE payment_status AS ENUM (
  'pending',              -- платёж создан, ожидает оплаты
  'succeeded',            -- оплата подтверждена
  'failed',               -- оплата не прошла
  'partially_refunded',   -- частичный возврат средств
  'refunded'              -- полный возврат средств
);

CREATE TYPE product_type AS ENUM (
  'entry_fee',     -- вступительный взнос (включает первый год)
  'subscription',  -- ежегодный членский взнос
  'event'          -- оплата участия в мероприятии
);

CREATE TYPE payment_provider AS ENUM (
  'yookassa',   -- ЮKassa
  'psb',        -- ПСБ Банк (прямое подключение СБП)
  'manual'      -- ручной платёж (внесён администратором)
);

CREATE TYPE change_status AS ENUM (
  'pending',   -- ожидает модерации
  'approved',  -- одобрено
  'rejected'   -- отклонено
);

CREATE TYPE event_status AS ENUM (
  'upcoming',   -- предстоящее
  'ongoing',    -- идёт сейчас
  'finished',   -- завершено
  'cancelled'   -- отменено
);

CREATE TYPE recording_status AS ENUM (
  'hidden',     -- скрыто
  'published'   -- опубликовано
);

CREATE TYPE video_source AS ENUM (
  'uploaded',   -- файл загружен в S3
  'external'    -- внешняя ссылка (YouTube, VK Video, Rutube и т.д.)
);

CREATE TYPE access_level AS ENUM (
  'public',            -- для всех
  'members_only',      -- только для членов ассоциации
  'participants_only'  -- только для участников мероприятия
);

CREATE TYPE notification_channel AS ENUM (
  'email',
  'telegram'
);

CREATE TYPE notification_status AS ENUM (
  'pending',  -- ожидает отправки
  'sent',     -- отправлено
  'failed'    -- ошибка отправки
);

CREATE TYPE document_type AS ENUM (
  'medical_diploma',   -- диплом о высшем медицинском образовании
  'retraining_cert',   -- сертификат о переподготовке
  'oncology_cert',     -- сертификат онколога
  'additional_cert'    -- дополнительный сертификат
);

CREATE TYPE article_status AS ENUM (
  'draft',      -- черновик
  'published',  -- опубликовано
  'archived'    -- в архиве
);

CREATE TYPE voting_session_status AS ENUM (
  'draft',      -- черновик
  'active',     -- голосование открыто
  'closed',     -- голосование закрыто
  'cancelled'   -- отменено
);

CREATE TYPE receipt_status AS ENUM (
  'pending',    -- чек запрошен
  'succeeded',  -- чек подтверждён
  'failed'      -- ошибка формирования
);

CREATE TYPE certificate_type AS ENUM (
  'member',   -- сертификат члена ассоциации
  'event'    -- сертификат участника мероприятия
);

CREATE TYPE receipt_type AS ENUM (
  'payment',    -- чек оплаты
  'refund'      -- чек возврата
);

CREATE TYPE event_registration_status AS ENUM (
  'pending',     -- ожидает оплаты
  'confirmed',   -- оплачено, подтверждено
  'cancelled'    -- отменено
);

CREATE TYPE role_type AS ENUM (
  'admin',
  'manager',
  'accountant',
  'doctor',
  'user'
);
```

## 14. Бизнес-правила на уровне БД

### 14.1 CHECK-ограничения

```sql
-- Платежи: сумма положительная
ALTER TABLE payments ADD CONSTRAINT chk_payments_amount_positive
  CHECK (amount > 0);

-- Чеки: сумма положительная
ALTER TABLE receipts ADD CONSTRAINT chk_receipts_amount_positive
  CHECK (amount > 0);

-- Тарифы мероприятий: цены неотрицательные, лимит мест
ALTER TABLE event_tariffs ADD CONSTRAINT chk_tariffs_price
  CHECK (price >= 0);
ALTER TABLE event_tariffs ADD CONSTRAINT chk_tariffs_member_price
  CHECK (member_price >= 0);
ALTER TABLE event_tariffs ADD CONSTRAINT chk_tariffs_seats_limit
  CHECK (seats_limit IS NULL OR seats_limit > 0);
ALTER TABLE event_tariffs ADD CONSTRAINT chk_tariffs_seats_taken
  CHECK (seats_taken >= 0);

-- Подписки: дата окончания после начала
ALTER TABLE subscriptions ADD CONSTRAINT chk_subs_dates
  CHECK (starts_at IS NULL OR ends_at IS NULL OR ends_at > starts_at);

-- Голосования: дата окончания после начала
ALTER TABLE voting_sessions ADD CONSTRAINT chk_voting_dates
  CHECK (ends_at > starts_at);

-- Сертификаты: год в разумных пределах
ALTER TABLE certificates ADD CONSTRAINT chk_certs_year
  CHECK (year >= 2020 AND year <= 2100);

-- Регистрации: цена неотрицательная
ALTER TABLE event_registrations ADD CONSTRAINT chk_reg_price
  CHECK (applied_price >= 0);

-- Тарифы подписок: цена положительная, длительность положительная
ALTER TABLE plans ADD CONSTRAINT chk_plans_price
  CHECK (price > 0);
ALTER TABLE plans ADD CONSTRAINT chk_plans_duration
  CHECK (duration_months > 0);
```

### 14.2 Триггеры

```sql
-- 1. Автообновление updated_at
-- Применяется к: users, doctor_profiles, subscriptions, payments,
-- receipts, articles, organization_documents, pages_seo, events,
-- plans, notification_templates, site_settings

CREATE OR REPLACE FUNCTION trigger_set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Пример (аналогично для каждой таблицы с updated_at):
CREATE TRIGGER trg_users_updated_at
  BEFORE UPDATE ON users
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();


-- 2. Контроль seats_taken при регистрации на мероприятие
CREATE OR REPLACE FUNCTION trigger_event_registration_seats()
RETURNS TRIGGER AS $$
DECLARE
  v_limit INT;
  v_taken INT;
BEGIN
  IF TG_OP = 'INSERT' AND NEW.status = 'confirmed' THEN
    SELECT seats_limit, seats_taken INTO v_limit, v_taken
      FROM event_tariffs WHERE id = NEW.event_tariff_id FOR UPDATE;
    IF v_limit IS NOT NULL AND v_taken >= v_limit THEN
      RAISE EXCEPTION 'No seats available for this tariff';
    END IF;
    UPDATE event_tariffs SET seats_taken = seats_taken + 1
      WHERE id = NEW.event_tariff_id;
  END IF;

  IF TG_OP = 'UPDATE'
     AND OLD.status = 'confirmed' AND NEW.status = 'cancelled' THEN
    UPDATE event_tariffs SET seats_taken = GREATEST(seats_taken - 1, 0)
      WHERE id = OLD.event_tariff_id;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_event_registration_seats
  AFTER INSERT OR UPDATE ON event_registrations
  FOR EACH ROW EXECUTE FUNCTION trigger_event_registration_seats();


-- 3. Деактивация профиля при истечении подписки
-- При обновлении subscriptions.status → 'expired' устанавливает
-- doctor_profiles.status = 'deactivated'
CREATE OR REPLACE FUNCTION trigger_subscription_expired()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.status = 'expired' AND OLD.status != 'expired' THEN
    UPDATE doctor_profiles
      SET status = 'deactivated'
      WHERE user_id = NEW.user_id
        AND status = 'active';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_subscription_expired
  AFTER UPDATE ON subscriptions
  FOR EACH ROW EXECUTE FUNCTION trigger_subscription_expired();


-- 4. Проверка что сессия голосования активна при голосовании
CREATE OR REPLACE FUNCTION trigger_vote_session_active()
RETURNS TRIGGER AS $$
DECLARE
  v_status voting_session_status;
BEGIN
  SELECT status INTO v_status
    FROM voting_sessions WHERE id = NEW.voting_session_id;
  IF v_status != 'active' THEN
    RAISE EXCEPTION 'Voting session is not active (status: %)', v_status;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_vote_session_active
  BEFORE INSERT ON votes
  FOR EACH ROW EXECUTE FUNCTION trigger_vote_session_active();
```

### 14.4 Soft Delete

В проекте soft delete (`deleted_at TIMESTAMPTZ DEFAULT NULL`) **не используется**. Вместо этого применяются:
- `is_active` / `status` для деактивации записей без удаления (users, cities, plans, certificates)
- `status = 'archived'` для контента (articles)
- `status = 'cancelled'` для отменённых сущностей (subscriptions, events, event_registrations)

Фактическое удаление (DELETE) допускается только для:
- Черновиков (`doctor_profile_changes` в статусе `pending` — при повторной отправке)
- Фото галерей (`event_gallery_photos`)
- Отдельных документов (`doctor_documents` при замене)

| # | Правило | Реализация |
